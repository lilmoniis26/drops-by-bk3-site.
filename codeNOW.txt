  <script>
    // Clock
    function updateClock() {
      const el = document.getElementById("hudClock");
      if (!el) return;
      const now = new Date();
      const hh = String(now.getHours()).padStart(2, "0");
      const mm = String(now.getMinutes()).padStart(2, "0");
      const ss = String(now.getSeconds()).padStart(2, "0");
      el.textContent = hh + ":" + mm + ":" + ss;
    }
    setInterval(updateClock, 1000);
    updateClock();

    // Nav switching (soft, just highlights + scrolls)
    const navItems = document.querySelectorAll(".nav-item");
    const panels = document.querySelectorAll("[data-section-panel]");

    navItems.forEach((item) => {
      item.addEventListener("click", () => {
        const target = item.getAttribute("data-section");
        navItems.forEach((n) => n.setAttribute("data-active", "false"));
        item.setAttribute("data-active", "true");

        // Soft behavior: scroll to relevant panel
        panels.forEach((panel) => {
          const sections = panel
            .getAttribute("data-section-panel")
            .split(" ")
            .filter(Boolean);
          if (sections.includes(target)) {
            panel.scrollIntoView({ behavior: "smooth", block: "center" });
          }
        });
      });
    });

    // Live widget simulation
    const toggleBtn = document.getElementById("toggleLiveBtn");
    const widgetStatus = document.getElementById("widgetStatus");
    const widgetStatusLabel = document.getElementById("widgetStatusLabel");
    const widgetScenePill = document.getElementById("widgetScenePill");
    const metricViewers = document.getElementById("metricViewers");
    const metricViewersFill = document.getElementById("metricViewersFill");
    const metricEnergy = document.getElementById("metricEnergy");
    const metricEnergyFill = document.getElementById("metricEnergyFill");
    const metricDrop = document.getElementById("metricDrop");

    let live = false;

    function setMeterWidth(el, pct) {
      if (!el) return;
      el.style.width = pct + "%";
    }

    toggleBtn.addEventListener("click", () => {
      live = !live;
      widgetStatus.dataset.live = live ? "true" : "false";
      widgetStatusLabel.textContent = live ? "LIVE // TACTICAL" : "OFFLINE STANDBY";
      widgetScenePill.textContent = live ? "Scene: Live Ops" : "Scene: Standby";

      if (live) {
        const viewers = Math.floor(Math.random() * 120) + 12;
        const energy = Math.floor(Math.random() * 40) + 60;
        const drops = ["CORE", "BUNDLE", "SIGNAL", "HEADSHOT"];
        const drop = drops[Math.floor(Math.random() * drops.length)];

        metricViewers.textContent = viewers;
        metricEnergy.textContent = energy + "%";
        metricDrop.textContent = drop;

        setMeterWidth(metricViewersFill, Math.min(viewers, 100));
        setMeterWidth(metricEnergyFill, energy);
      } else {
        metricViewers.textContent = "0";
        metricEnergy.textContent = "0%";
        metricDrop.textContent = "CORE";
        setMeterWidth(metricViewersFill, 10);
        setMeterWidth(metricEnergyFill, 10);
      }
    });

    // Initial meter state
    setMeterWidth(metricViewersFill, 10);
    setMeterWidth(metricEnergyFill, 10);
